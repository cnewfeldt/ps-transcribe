**1. Create AnalysisPanel.swift** at `PSTranscribe/Sources/PSTranscribe/Views/AnalysisPanel.swift`.

Follow the UI-SPEC component inventory exactly. The panel takes AnalysisState (for live mode) and an optional ParsedAnalysis (for review mode):

```swift
import SwiftUI

struct AnalysisPanel: View {
    let summary: String
    let actionItems: [String]
    let keyTopics: [String]
    let isUpdating: Bool
    let isLive: Bool  // true during recording, false for review mode
    let hasData: Bool

    var body: some View {
        VStack(spacing: 0) {
            // Panel header
            HStack {
                Text("Analysis")
                    .font(.system(size: 10, weight: .semibold))
                    .textCase(.uppercase)
                    .tracking(0.8)
                    .foregroundStyle(Color.fg2)
                Spacer()
                if isLive && isUpdating {
                    // Pulsing accent1 4pt dot per UI-SPEC
                    Circle()
                        .fill(Color.accent1)
                        .frame(width: 4, height: 4)
                        .opacity(/* pulsing */)
                }
            }
            .padding(.horizontal, 16)
            .padding(.vertical, 8)

            Divider()

            if !hasData && !isUpdating {
                // Empty state per UI-SPEC
                emptyState
            } else if !hasData && isUpdating {
                // Loading state per UI-SPEC
                loadingState
            } else {
                // Live/review state with section cards
                ScrollView {
                    VStack(spacing: 24) {
                        if !summary.isEmpty {
                            analysisSectionCard(title: "Summary") {
                                Text(summary)
                                    .font(.system(size: 12))
                                    .foregroundStyle(Color.fg1)
                            }
                        }
                        if !actionItems.isEmpty {
                            analysisSectionCard(title: "Action Items") {
                                VStack(alignment: .leading, spacing: 8) {
                                    ForEach(actionItems, id: \.self) { item in
                                        HStack(alignment: .top, spacing: 6) {
                                            Image(systemName: "square")
                                                .font(.system(size: 12))
                                                .foregroundStyle(Color.fg3)
                                            Text(item)
                                                .font(.system(size: 12))
                                                .foregroundStyle(Color.fg1)
                                                .fixedSize(horizontal: false, vertical: true)
                                        }
                                    }
                                }
                            }
                        }
                        if !keyTopics.isEmpty {
                            analysisSectionCard(title: "Key Topics") {
                                // Wrapping pills layout
                                FlowLayout(spacing: 6) {
                                    ForEach(keyTopics, id: \.self) { topic in
                                        Text(topic)
                                            .font(.system(size: 10, weight: .semibold))
                                            .foregroundStyle(Color.fg2)
                                            .padding(.horizontal, 8)
                                            .padding(.vertical, 4)
                                            .background(Color.bg0.opacity(0.8))
                                            .clipShape(Capsule())
                                            .overlay(Capsule().stroke(Color.white.opacity(0.08)))
                                    }
                                }
                            }
                        }
                    }
                    .padding(16)
                }
            }
        }
        .frame(minWidth: 220, idealWidth: 260, maxWidth: 300)
        .background(Color.bg2)
        .overlay(Divider(), alignment: .leading)
    }

    // ... empty state, loading state, analysisSectionCard helper views
}
```

For the pulsing dot, use a simple `@State private var pulse = false` with `.animation(.easeInOut(duration: 0.9).repeatForever(autoreverses: true), value: pulse)` matching the existing PulsingDot pattern in ControlBar.swift.

For `FlowLayout` -- since macOS 26 has Layout protocol, implement a simple wrapping layout, OR use a simpler approach: a `LazyVGrid` with adaptive columns. If Layout is too complex, use `VStack` with `.fixedSize()` wrapping -- key topics as a simple comma-joined text is also acceptable and simpler. The UI-SPEC shows both options. Use the simplest approach that looks right.

Follow the AnalysisSectionCard structure from UI-SPEC exactly:
- Section heading: 10pt semibold uppercase tracking:0.8 accent1
- Left accent stripe: Rectangle 3pt wide, accent1 opacity 0.4
- Content area: padding(.leading, 8)
- Card: padding 16, bg1 opacity 0.7, cornerRadius 10, white stroke 0.06
