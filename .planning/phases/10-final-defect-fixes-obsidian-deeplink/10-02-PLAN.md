```swift
/// Derives the Obsidian vault name from a vault subfolder path.
/// Given "/Users/cary/Documents/MyVault/Meetings", returns "MyVault".
/// Returns nil if the path is empty or has no meaningful parent.
func obsidianVaultName(from vaultSubPath: String) -> String? {
    guard !vaultSubPath.isEmpty else { return nil }
    let url = URL(fileURLWithPath: vaultSubPath)
    let name = url.deletingLastPathComponent().lastPathComponent
    // Guard against root paths where lastPathComponent is "/" or empty
    guard !name.isEmpty, name != "/" else { return nil }
    return name
}

/// Constructs an obsidian://open URL for a transcript file.
/// - Parameters:
///   - filePath: Absolute path to the transcript file on disk
///   - vaultRoot: Absolute path to the Obsidian vault root directory
///   - vaultName: Name of the Obsidian vault (typically vaultRoot's lastPathComponent)
/// - Returns: An obsidian://open URL with vault and file query parameters, or nil if inputs are invalid
func makeObsidianURL(filePath: String, vaultRoot: String, vaultName: String) -> URL? {
    guard !filePath.isEmpty,
          !vaultRoot.isEmpty,
          !vaultName.isEmpty,
          filePath.hasPrefix(vaultRoot) else { return nil }

    // Strip vault root prefix and leading separator to get vault-relative path
    var relative = String(filePath.dropFirst(vaultRoot.count))
    if relative.hasPrefix("/") { relative = String(relative.dropFirst()) }
    guard !relative.isEmpty else { return nil }

    var components = URLComponents()
    components.scheme = "obsidian"
    components.host = "open"
    components.queryItems = [
        URLQueryItem(name: "vault", value: vaultName),
        URLQueryItem(name: "file", value: relative)
    ]
    return components.url
}
```

**Important:** `URLQueryItem` will percent-encode the values automatically via `URLComponents`. If the `/` in file paths gets encoded as `%2F` and Obsidian doesn't handle that, the manual verification checkpoint in Task 3 will catch it. The `URLComponents` approach is correct per Apple docs and handles spaces, special chars properly.

Run `swift test --filter ObsidianURLTests` -- all tests must pass.

```swift
private func obsidianURLForEntry(_ entry: LibraryEntry) -> URL? {

    let vaultSubPath: String

    return makeObsidianURL(filePath: entry.filePath, vaultRoot: vaultRoot, vaultName: vaultName)
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User vault path -> Obsidian URL | File path from LibraryEntry compared against vault paths to construct external URL |
| App -> External app (Obsidian) | NSWorkspace.shared.open() launches Obsidian with a custom URL scheme |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-03 | Tampering | makeObsidianURL file path | mitigate | Guard `filePath.hasPrefix(vaultRoot)` before constructing URL -- prevents path traversal or arbitrary file references in the obsidian:// URL |
| T-10-04 | Tampering | vault name derivation | accept | Vault name is derived programmatically from the user's own configured vault path, not from external input. No injection vector. |
| T-10-05 | Information Disclosure | obsidian:// URL | accept | URL contains vault name and relative file path -- both are local-only data, not transmitted over network. Obsidian processes locally. |
| T-10-06 | Spoofing | NSWorkspace.shared.open | accept | macOS handles URL scheme dispatch. If a malicious app registers obsidian://, that's a system-level concern outside app scope. Standard macOS app behavior. |
</threat_model>
